from __future__ import division
from pylab import *

N = 16
x = linspace(-1, 1, num=N)
y = linspace(-1, 1, num=N)
x, y = meshgrid(x, y)
z = x**2 + pow(y, 2) # ** and pow(y, 2) are equivalent

figure()
imshow(z, interpolation='nearest') # interpolation for no fuzziness
colorbar()
savefig('z.png', dpi=300)
show()
